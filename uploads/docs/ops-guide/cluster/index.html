    集群部署

    OpsAny社区版高可用集群部署¶

    推荐使用Openresty+keepalived实现负载均衡和高可用架构，默认OpsAny社区版已经使用了Openresty而且也是使用了负载均衡的配置模式，即便只有一台主机，现在你需要部署Keepalived并且 修改Openresty的配置并重启opsany-openresty即可。

    部署多个节点¶

    使用脚本增加部署一个节点，注意MySQL、Redis、MongoDB、RabbitMQ均使用先部署的节点，请自行创建MySQL主从、Redis集群或者主从、MongoDB机器和RabbitMQ集群。

    这个是主要的工作量，服务的部署反而是最简单的。

    部署Keepalived¶

    1.在所有集群节点上都部署Keepalived

    推荐3台主机做集群，最少2台主机，保证高可用。

    [root@linux-node1 ~]# yum install -y keepalived
[root@linux-node2 ~]# yum install -y keepalived

    2.配置Keepavlied

    推荐试用不抢占模式，所有Keepalived都设置为BACKUP状态，通过priority进行选举。

    linux-node1配置：

    [root@linux-node1 ~]# vim /etc/keepalived/keepalived.conf
! Configuration File for keepalived
global_defs {
   notification_email {
     k8s@example.com
   }
   notification_email_from keepalived@example.com
   smtp_server 127.0.0.1
   smtp_connect_timeout 30
   router_id opsany-ha
}

vrrp_script haproxy_health_check {
    script "/etc/keepalived/opsany_health_check.sh"
    interval 3
    weight -2
    fall 10
    rise 2
}

vrrp_instance opsany_ha {
    state BACKUP
    nopreempt
    interface eth0
    virtual_router_id 36
    priority 150
    advert_int 1
    authentication {
        auth_type PASS
        auth_pass 57459267
    }
    virtual_ipaddress {
       192.168.56.20
    }
    track_script {
        haproxy_health_check
    }
}

    linux-node2配置：

    [root@linux-node1 ~]# vim /etc/keepalived/keepalived.conf

    3.启动并测试

    #linux-node1
[root@linux-node1 ~]# systemctl enable --now keepalived
[root@linux-node1 ~]# ip ad li

#linux-node2
[root@linux-node2 ~]# systemctl enable --now keepalived
[root@linux-node2 ~]# ip ad li

    配置OpenResty¶

    OpsAny默认试用Openresty配置文件是挂载到容器中的，修改完毕配置文件之后，重启容器即可。

    [root@linux-node1 ~]# vim /opt/opsany/conf/nginx-conf.d/nginx_paas.conf
# upstream - web
upstream OPEN_PAAS {
    server 192.168.56.11:8001 max_fails=1  fail_timeout=30s;
}
upstream OPEN_PAAS_APPENGINE {
    server 192.168.56.11:8000 max_fails=1  fail_timeout=30s;
}
upstream OPEN_PAAS_ESB {
    server 192.168.56.11:8002 max_fails=1  fail_timeout=30s;
}
upstream OPEN_PAAS_LOGIN {
    server 192.168.56.11:8003 max_fails=1  fail_timeout=30s;
}
upstream CONTROL_WS {
    server 192.168.56.11:8004 max_fails=1  fail_timeout=30s;
}
upstream MONITOR_ZABBIX {
    server 192.168.56.11:8006 max_fails=1  fail_timeout=30s;
}
upstream DASHBOARD_GRAFANA {
    server 192.168.56.11:8007 max_fails=1  fail_timeout=30s;
}

# upstream - paasagent
upstream PAAS_AGENT_TEST {
    server 192.168.56.11:8084 max_fails=1  fail_timeout=30s;
}
upstream PAAS_AGENT_PROD {
    server 192.168.56.11:8085 max_fails=1  fail_timeout=30s;
}
# proxy_next_upstream  http_502 http_504 error timeout invalid_header;

server {
    listen 80;
    server_name 192.168.56.11;
    location ~ ^/uploads/(.*) {
        autoindex off;
        root /opt/opsany/;
    }
    location ~/ {
        rewrite ^(.*)$ https://$host$1 permanent;
    }
}

server {
     listen       443 ssl;
     server_name  192.168.56.11;
     access_log /opt/opsany/logs/paas_nginx_access.log;
     error_log /opt/opsany/logs/paas_nginx_error.log;
     ssl_certificate /etc/nginx/conf.d/ssl/192.168.56.11.pem;
     ssl_certificate_key /etc/nginx/conf.d/ssl/192.168.56.11.key;
     ssl_session_timeout 5m;
     ssl_ciphers ECDHE-RSA-AES128-GCM-SHA256:ECDHE:ECDH:AES:HIGH:!NULL:!aNULL:!MD5:!ADH:!RC4;
     ssl_protocols TLSv1 TLSv1.1 TLSv1.2;
     ssl_prefer_server_ciphers on;
    # gzip config
    gzip on;
    gzip_min_length 1k;
    gzip_comp_level 9;
    gzip_types text/plain application/javascript application/x-javascript text/css application/xml text/javascript application/x-httpd-php image/jpeg image/gif image/png;
    gzip_vary on;
    gzip_disable "MSIE [1-6]\.";
    client_max_body_size    512m;

    # ============================ paas ============================
    # Zabbix
    location /zabbix/ {
        proxy_pass http://MONITOR_ZABBIX;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_redirect off;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Host $server_name;
    }

    # Grafana
    location /grafana/ {
        proxy_pass https://DASHBOARD_GRAFANA;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_redirect off;
        proxy_set_header Host $host;
        proxy_set_header X-WEBAUTH-USER admin;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Host $server_name;
    }

    # CONTROL WebSocket
     location /ws/control/ {
        proxy_pass http://CONTROL_WS;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_redirect off;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Host $server_name;
    }
    # PAAS_SERVICE HOST/PORT
    location / {
        proxy_pass http://OPEN_PAAS;
        proxy_pass_header Server;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Scheme $scheme;
        proxy_set_header Host $http_host;
        proxy_redirect off;
        proxy_read_timeout 600;
    }

    # uploads
    location ~ ^/uploads/(.*) {
        autoindex off;
        root /opt/opsany/;
    }

    # docs
    location ~ ^/docs/ {
        index index.html;
        root /opt/opsany/uploads/;
    }

    # ============================ appengine ============================
    # ENGINE_SERVICE HOST/PORT
    location ~ ^/v1 {
        proxy_pass http://OPEN_PAAS_APPENGINE;
        proxy_pass_header Server;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header Host $http_host;
        proxy_set_header X-Scheme $scheme;
        proxy_read_timeout 600;
    }

    # ============================ esb ============================
    # ESB_SERVICE HOST/PORT
    location ~ ^/api/(.*) {
        proxy_pass http://OPEN_PAAS_ESB/$1$is_args$args;
        proxy_pass_header Server;
        proxy_set_header X-Request-Uri $request_uri;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Scheme $scheme;
        proxy_set_header Host $http_host;
        proxy_redirect off;
        proxy_read_timeout 600;
    }


    # ============================ login ============================
    # LOGIN_SERVICE HOST/PORT
    location ~ ^/login/(.*) {
        proxy_pass http://OPEN_PAAS_LOGIN/$1$is_args$args;
        proxy_pass_header Server;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Scheme $scheme;
        proxy_set_header Host $http_host;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_redirect off;
        proxy_read_timeout 600;
    }


    # ============================ paas_agent ============================
    # for apps test
    location ~ ^/t/ {
        proxy_pass http://PAAS_AGENT_TEST;
        proxy_pass_header Server;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Scheme $scheme;
        proxy_set_header Host $http_host;
        proxy_redirect off;
        proxy_read_timeout 600;
    }

    # for apps prod
    location ~ ^/o/ {
        proxy_pass http://PAAS_AGENT_PROD;
        proxy_pass_header Server;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Scheme $scheme;
        proxy_set_header Host $http_host;
        proxy_redirect off;
        proxy_read_timeout 600;
    }
